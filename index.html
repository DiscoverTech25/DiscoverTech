import React from 'react';

import { Card, CardContent } from '@/components/ui/card';

import { Button } from '@/components/ui/button';

import { Input } from '@/components/ui/input';

import { Textarea } from '@/components/ui/textarea';

import { useState } from 'react';


const Home = () => (

  <section className='p-10 bg-green-100 min-h-screen'>

    <h1 className='text-4xl font-bold text-green-800 mb-4'>Welcome to Our STEM Education Initiative</h1>

    <p className='text-lg text-green-700'>Empowering the next generation with free engineering and coding classes.</p>

  </section>

);


const AboutUs = () => (

  <section className='p-10 bg-white min-h-screen'>

    <h2 className='text-3xl font-bold text-green-800 mb-4'>About Us</h2>

    <p className='text-lg text-gray-700'>We are dedicated to teaching STEM and engineering classes, including an Introduction to Python, at libraries for free. Our mission is to make coding accessible and fun for young learners.</p>

  </section>

);


const Locations = () => (

  <section className='p-10 bg-green-50 min-h-screen'>

    <h2 className='text-3xl font-bold text-green-800 mb-4'>Locations</h2>

    <p className='text-lg text-gray-700'>We currently offer classes at various library locations. Check back for updates!</p>

  </section>

);


const Volunteering = () => {

  const [formData, setFormData] = useState({ name: '', email: '', message: '' });

  const handleChange = (e) => setFormData({ ...formData, [e.target.name]: e.target.value });

  const handleSubmit = (e) => {

    e.preventDefault();

    alert('Thank you for signing up to volunteer!');

  };


  return (

    <section className='p-10 bg-white min-h-screen'>

      <h2 className='text-3xl font-bold text-green-800 mb-4'>Volunteering</h2>

      <p className='text-lg text-gray-700'>Join us in making a difference! Sign up to volunteer today.</p>

      <form onSubmit={handleSubmit} className='mt-6'>

        <Input type='text' name='name' placeholder='Your Name' value={formData.name} onChange={handleChange} className='mb-4' required />

        <Input type='email' name='email' placeholder='Your Email' value={formData.email} onChange={handleChange} className='mb-4' required />

        <Textarea name='message' placeholder='Why do you want to volunteer?' value={formData.message} onChange={handleChange} className='mb-4' required />

        <Button type='submit'>Submit</Button>

      </form>

    </section>

  );

};


const Contact = () => (

  <section className='p-10 bg-green-50 min-h-screen'>

    <h2 className='text-3xl font-bold text-green-800 mb-4'>Contact Us</h2>

    <p className='text-lg text-gray-700'>Have questions or want to learn more? Reach out to us!</p>

  </section>

);


const App = () => (

  <div className='font-sans text-gray-800'>

    <Home />

    <AboutUs />

    <Locations />

    <Volunteering />

    <Contact />

  </div>

);


export default App;
